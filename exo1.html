Requete 1

Liste de differents personne travaillant sur les robots

SELECT ?nom_de_famille ?nom_de_familleLabel ?robot ?domaine_d_activité ?domaine_d_activitéLabel ?lieu_de_travail ?lieu_de_travailLabel ?occupation ?occupationLabel ?image WHERE { SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". } ?robot wdt:P101 wd:Q11012. OPTIONAL { ?robot wdt:P101 ?domaine_d_activité. } OPTIONAL { ?robot wdt:P735 ?prénom. } OPTIONAL { ?robot wdt:P937 ?lieu_de_travail. } OPTIONAL { ?robot wdt:P106 ?occupation. } OPTIONAL { ?robot wdt:P734 ?nom_de_famille. } OPTIONAL { ?robot wdt:P18 ?image. } } LIMIT 100
